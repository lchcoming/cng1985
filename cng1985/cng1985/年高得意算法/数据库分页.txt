
ALTER PROCEDURE dbo.PageReTalk
(
@page int,
@size int,
@cur int
)
	/*
	(
	@parameter1 int = 5,
	@parameter2 datatype OUTPUT
	)
	*/
AS
	/* SET NOCOUNT ON */
	
	SELECT   TOP (@size) ID, [Content], Pubtime, TalkID
FROM      ReTalk
WHERE   ID NOT IN
                    (SELECT   TOP (@size*(@page-1)) ID
                     FROM    ReTalk
                     where TalkID = @cur
                     )
                     and TalkID = @cur
                     
                     	RETURN
